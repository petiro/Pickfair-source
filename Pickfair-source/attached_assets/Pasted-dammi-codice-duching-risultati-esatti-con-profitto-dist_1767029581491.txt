тЬЕ Codice Python (BACK)
def dutching_back(odds, total_stake, commission=0.05):
    """
    odds: list[float]
    total_stake: float
    return: list of dict with stake & profit
    """

    inv_sum = sum(1 / (o - 1) for o in odds)

    results = []
    profits = []

    for o in odds:
        stake = total_stake / inv_sum / (o - 1)
        profit = stake * (o - 1)
        if profit > 0:
            profit *= (1 - commission)

        results.append({
            "odds": round(o, 3),
            "stake": round(stake, 2),
            "profit": round(profit, 2)
        })
        profits.append(profit)

    return results, round(min(profits), 2)

ЁЯзк Esempio

print(res)
print("Profitto uniforme:", uniform_profit)

тЬЕ Codice Python (LAY)
def dutching_lay(odds, target_profit, commission=0.05):
    """
    odds: list[float]
    target_profit: profitto netto uniforme desiderato
    """

    results = []

    for o in odds:
        stake = target_profit / (o - 1)
        liability = stake * (o - 1)

        profit = target_profit
        if profit > 0:
            profit *= (1 - commission)

        results.append({
            "odds": round(o, 3),
            "lay_stake": round(stake, 2),
            "liability": round(liability, 2),
            "profit": round(profit, 2)
        })

    return results
