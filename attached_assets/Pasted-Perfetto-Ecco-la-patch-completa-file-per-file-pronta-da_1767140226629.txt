from PyQt5.QtWidgets import QWidget, QTableWidget, QTableWidgetItem
from PyQt5.QtCore import Qt, pyqtSignal

class LadderDragWidget(QWidget):
    pnl_updated = pyqtSignal(str, float)  # selection_id, pnl

    def __init__(self, market_id, selections):
        super().__init__()
        self.market_id = market_id
        self.selections = selections
        self.table = QTableWidget(len(selections), 7, self)
        self.table.setHorizontalHeaderLabels(["Selection","Back1","Back2","Lay1","Lay2","Stake","P/L"])
        self.table.setDragDropMode(QTableWidget.InternalMove)
        self.populate()

    def populate(self):
        for row, sel in enumerate(self.selections):
            self.table.setItem(row, 0, QTableWidgetItem(sel["name"]))
            self.table.setItem(row, 1, QTableWidgetItem(str(sel["back"][0])))
            self.table.setItem(row, 2, QTableWidgetItem(str(sel["back"][1])))
            self.table.setItem(row, 3, QTableWidgetItem(str(sel["lay"][0])))
            self.table.setItem(row, 4, QTableWidgetItem(str(sel["lay"][1])))
            self.table.setItem(row, 5, QTableWidgetItem(str(sel["stake"])))
            self.table.setItem(row, 6, QTableWidgetItem(str(sel.get("pnl",0.0))))
        self.table.resizeColumnsToContents()

    def update_pnl(self, selection_id, pnl):
        for row, sel in enumerate(self.selections):
            if sel["id"] == selection_id:
                sel["pnl"] = pnl
                self.table.item(row, 6).setText(str(round(pnl,2)))
                self.pnl_updated.emit(selection_id, pnl)
                break

2️⃣ ai/auto_scalp.py – AI Auto-Scalp Pattern-Based
Copia codice

import numpy as np
from sklearn.ensemble import RandomForestClassifier
from collections import deque

class AutoScalpAI:
    def __init__(self, window_size=20):
        self.window_size = window_size
        self.history = {}  # selection_id -> deque
        self.model = RandomForestClassifier(n_estimators=100)
        self.model_trained = False

    def push_tick(self, selection_id, tick):
        if selection_id not in self.history:
            self.history[selection_id] = deque(maxlen=self.window_size)
        self.history[selection_id].append(tick)

    def predict_signal(self, selection_id):
        if not self.model_trained or selection_id not in self.history:
            return None
        ticks = self.history[selection_id]
        if len(ticks) < self.window_size:
            return None
        X = np.array(ticks).reshape(1, -1)
        return self.model.predict(X)[0]  # 1=BACK, -1=LAY, 0=HOLD

    def train(self, X_train, y_train):
        self.model.fit(X_train, y_train)
        self.model_trained = True

from ai.auto_scalp import AutoScalpAI
from automation.trailing_engine import TrailingStopEngine
from handlers.partial_green import partial_green

auto_scalp = AutoScalpAI()
trailing_engine = TrailingStopEngine(trail_amount=0.2)

def process_tick(selection_id, tick, order, ladder_widget):
    auto_scalp.push_tick(selection_id, tick)
    signal = auto_scalp.predict_signal(selection_id)

    current_pnl = order_manager.get_pnl(order["betId"])
    
    # Trailing stop + auto-green
    if trailing_engine.update(order["betId"], current_pnl):
        partial_green(order, percent=0.5)
        trailing_engine.reset(order["betId"])

    # Auto-scalp
    if signal == 1:
        order_manager.place_back(selection_id, stake=order["stake"], price=tick["best_back"])
    elif signal == -1:
        order_manager.place_lay(selection_id, stake=order["stake"], price=tick["best_lay"])

    # Aggiorna UI P&L live
    ladder_widget.update_pnl(selection_id, current_pnl)

from PyQt5.QtWidgets import QWidget, QTableWidget, QTableWidgetItem

class MarketWatchWidget(QWidget):
    def __init__(self, markets):
        super().__init__()
        self.markets = markets
        self.table = QTableWidget(len(markets), 10, self)
        self.table.setHorizontalHeaderLabels(["Market","In-play","Status","Min P/L","Max P/L","Start Time","P.Bets","U.Bets","M.Bets","Last Refresh"])
        self.populate()

    def populate(self):
        for row, mkt in enumerate(self.markets):
            self.table.setItem(row, 0, QTableWidgetItem(mkt["name"]))
            self.table.setItem(row, 1, QTableWidgetItem(str(mkt["in_play"])))
            self.table.setItem(row, 2, QTableWidgetItem(mkt["status"]))
            self.table.setItem(row, 3, QTableWidgetItem(str(round(mkt["min_pnl"],2))))
            self.table.setItem(row, 4, QTableWidgetItem(str(round(mkt["max_pnl"],2))))
            self.table.setItem(row, 5, QTableWidgetItem(mkt["start_time"]))
            self.table.setItem(row, 6, QTableWidgetItem(str(mkt["pending_bets"])))
            self.table.setItem(row, 7, QTableWidgetItem(str(mkt["unmatched_bets"])))
            self.table.setItem(row, 8, QTableWidgetItem(str(mkt["matched_bets"])))
            self.table.setItem(row, 9, QTableWidgetItem(mkt["last_refresh"]))
        self.table.resizeColumnsToContents()

    def update_pnl(self, market_id, min_pnl, max_pnl):
        for row, mkt in enumerate(self.markets):
            if mkt["id"] == market_id:
                self.table.item(row, 3).setText(str(round(min_pnl,2)))
                self.table.item(row, 4).setText(str(round(max_pnl,2)))
                break
