def validate_selections(results):
    for r in results:
        if r["stake"] < MIN_STAKE:
            raise ValueError(f"Stake troppo basso: {r['stake']}")

        win = r.get("profit_if_win", 0)
        if win > MAX_WIN:
            raise ValueError(f"Vincita massima superata: {win}")

def calculate_dutching_stakes(
    odds,
    total_stake,
    side="BACK",
    commission=0.05
):
    if side == "BACK":
        results = _calculate_back_dutching(odds, total_stake, commission)
    elif side == "LAY":
        results = _calculate_lay_dutching(odds, total_stake)
    else:
        raise ValueError("side deve essere BACK o LAY")

    validate_selections(results)
    return results

🔵 DUTCHING BACK – PROFITTO UNIFORME
Formula: 1 / (price - 1)
def _calculate_back_dutching(odds, total_stake, commission=0.05):
    inv_weights = [1 / (o - 1) for o in odds]
    weight_sum = sum(inv_weights)

    results = []
    profits = []

    for o, w in zip(odds, inv_weights):
        stake = total_stake * w / weight_sum
        gross_profit = stake * (o - 1)
        net_profit = gross_profit * (1 - commission)

        results.append({
            "side": "BACK",
            "odds": round(o, 3),
            "stake": round(stake, 2),
            "profit_if_win": round(net_profit, 2)
        })
        profits.append(net_profit)

    uniform_profit = round(min(profits), 2)
    for r in results:
        r["uniform_profit"] = uniform_profit

    return results

🔴 DUTCHING LAY – DISTRIBUZIONE PROPORZIONALE
def _calculate_lay_dutching(odds, total_liability):
    weights = [1 / (o - 1) for o in odds]
    weight_sum = sum(weights)

    results = []

    for o, w in zip(odds, weights):
        liability = total_liability * w / weight_sum
        stake = liability / (o - 1)

        results.append({
            "side": "LAY",
            "odds": round(o, 3),
            "stake": round(stake, 2),
            "liability": round(liability, 2),
            "profit_if_win": round(stake, 2)
        })

    return results

import numpy as np

def calculate_mixed_dutching(odds, sides, target_profit, commission=0.05):
    n = len(odds)
    A = []
    b = []

    for k in range(n):
        row = []
        for i in range(n):
            o = odds[i]
            if sides[i] == "BACK":
                row.append(o - 1 if i == k else -1)
            else:  # LAY
                row.append(-(o - 1) if i == k else 1)
        A.append(row)
        b.append(target_profit)

    stakes = np.linalg.solve(np.array(A), np.array(b))
    stakes = np.maximum(stakes, 0)

    results = []
    net_profit = target_profit * (1 - commission)

    for i in range(n):
        results.append({
            "side": sides[i],
            "odds": odds[i],
            "stake": round(stakes[i], 2),
            "uniform_profit": round(net_profit, 2)
        })

    validate_selections(results)
    return results

def adjust_for_slippage(matched_orders, remaining_odds, target_profit, commission=0.05):

        o["stake"] * (o["odds"] - 1) for o in matched_orders if o["side"] == "BACK"

🔁 SWAP DUTCHING MULTI-MARKET
def multi_market_swap(markets, target_profit):
    exposure = 0.0

    for m in markets:
        if m["side"] == "BACK":
            exposure += target_profit / (m["odds"] - 1)
        else:
            exposure += target_profit / m["odds"]

    return round(exposure, 2)

def dynamic_cashout(orders, live_odds, commission=0.05):
    profit = 0.0

    for o in orders:
        sel = o["selection"]
        live_price = live_odds[sel]

        if o["side"] == "BACK":
            hedge = (o["odds"] / live_price) * o["stake"]
            profit += o["stake"] * (o["odds"] - 1) - hedge
        else:  # LAY
            hedge = (live_price / o["odds"]) * o["stake"]
            profit += o["stake"] - hedge * (live_price - 1)

    if profit > 0:
        profit *= (1 - commission)

    return round(profit, 2)

def calculate_green_up(order, live_price, commission=0.05):
    if order["side"] == "BACK":
        hedge = (order["odds"] / live_price) * order["stake"]
        profit = order["stake"] * (order["odds"] - 1) - hedge
    else:
        hedge = (live_price / order["odds"]) * order["stake"]
        profit = order["stake"] - hedge * (live_price - 1)

    if profit > 0:
        profit *= (1 - commission)

    return round(profit, 2), round(hedge, 2)
