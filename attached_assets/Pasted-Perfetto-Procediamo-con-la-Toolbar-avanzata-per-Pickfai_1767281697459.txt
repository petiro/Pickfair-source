# ui/toolbar.py
import customtkinter as ctk

class Toolbar(ctk.CTkFrame):
    def __init__(self, parent, controller, *args, **kwargs):
        super().__init__(parent, *args, **kwargs)
        self.controller = controller
        self._draw_toolbar()

    def _draw_toolbar(self):
        # Simulation Toggle
        self.sim_toggle = ctk.CTkCheckBox(self, text="Simulation Mode",
                                           command=self.toggle_simulation)
        self.sim_toggle.grid(row=0, column=0, padx=4)

        # Auto-Green Toggle
        self.green_toggle = ctk.CTkCheckBox(self, text="Auto-Green",
                                            command=self.toggle_auto_green)
        self.green_toggle.grid(row=0, column=1, padx=4)

        # AI WoM Mixed Toggle
        self.ai_toggle = ctk.CTkCheckBox(self, text="AI Mixed Mode",
                                         command=self.toggle_ai)
        self.ai_toggle.grid(row=0, column=2, padx=4)

        # Status badge
        self.status_label = ctk.CTkLabel(self, text="Market OK ✅", fg_color="green")
        self.status_label.grid(row=0, column=3, padx=8)

        # Preset stake buttons
        for idx, pct in enumerate([25, 50, 100]):
            btn = ctk.CTkButton(self, text=f"{pct}%", width=50,
                                command=lambda p=pct: self.set_preset_stake(p))
            btn.grid(row=0, column=4+idx, padx=2)

    # --- Handlers ---
    def toggle_simulation(self):
        self.controller.simulation = self.sim_toggle.get()
        print(f"Simulation mode: {self.controller.simulation}")

    def toggle_auto_green(self):
        self.controller.auto_green_enabled = self.green_toggle.get()
        print(f"Auto-Green enabled: {self.controller.auto_green_enabled}")

    def toggle_ai(self):
        self.controller.ai_enabled = self.ai_toggle.get()
        print(f"AI Mixed enabled: {self.controller.ai_enabled}")

    def set_preset_stake(self, pct):
        self.controller.preset_stake_pct = pct / 100
        print(f"Preset stake set to {pct}%")

# controllers/dutching_controller.py
class DutchingController:
    def __init__(self, broker, pnl_engine, simulation=False):
        self.broker = broker
        self.pnl_engine = pnl_engine
        self.simulation = simulation
        self.auto_green_enabled = True
        self.ai_enabled = True
        self.preset_stake_pct = 1.0  # default 100%

    def submit_dutching(self, market_id, market_type, selections, total_stake=None, mode="BACK", ai_enabled=True):
        total_stake = total_stake or (self.preset_stake_pct * 10.0)  # fallback stake 10€
        if self.simulation:
            print("SIMULATION MODE - no real orders")
        # AI Mixed
        if ai_enabled and self.ai_enabled:
            selections = self._apply_ai_mixed(selections)
        # Submit orders (real or simulated)
        results = []
        for s in selections:
            stake = total_stake / len(selections)
            results.append({
                "selectionId": s['selectionId'],
                "runnerName": s['runnerName'],
                "side": s.get('side', mode),
                "stake": stake,
                "market_id": market_id,
                "market_type": market_type,
                "simulated": self.simulation,
                "auto_green": self.auto_green_enabled
            })
        return results

    def _apply_ai_mixed(self, selections):
        # Calcolo BACK/LAY automatico basato su WoM edge
        edge_scores = self.wom_engine.get_ai_edge_score(selections)
        for s in selections:
            s['side'] = edge_scores.get(s['selectionId'], {'side':'BACK'})['side']
        return selections
