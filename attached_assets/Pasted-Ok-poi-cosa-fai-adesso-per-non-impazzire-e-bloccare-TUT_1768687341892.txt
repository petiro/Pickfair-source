STEP 1) Metti un “guard” globale su OGNI click (log + timer)
Così quando clicchi un bottone e freeza, nel log vedi QUALE callback lo ha causato.
In main.py aggiungi questo:
Copia codice
Python
import time
import threading
import traceback
import logging

logger = logging.getLogger(__name__)

def is_ui_thread():
    return threading.current_thread() == threading.main_thread()

def ui_guard(name: str, warn_ms: int = 200):
    """
    Decoratore per callback UI.
    Logga durata e avvisa se supera warn_ms.
    """
    def deco(fn):
        def wrapper(*args, **kwargs):
            t0 = time.perf_counter()
            try:
                return fn(*args, **kwargs)
            finally:
                dt = (time.perf_counter() - t0) * 1000
                if dt > warn_ms:
                    logger.warning(f"[UI-GUARD] SLOW UI callback {name}: {dt:.0f}ms")
        return wrapper
    return deco

STEP 2) Blocca per sempre le API nel main thread (hard rule)
Questo è il fix che ti salva la vita.
Aggiungi questo decoratore:
Copia codice
Python
def assert_not_ui_thread(name: str = ""):
    def deco(fn):
        def wrapper(*args, **kwargs):
            if is_ui_thread():
                msg = f"[BUG] API called in UI thread: {name or fn.__name__}"
                logger.error(msg)
                raise RuntimeError(msg)
            return fn(*args, **kwargs)
        return wrapper
    return deco

STEP 3) Watchdog che quando freeza fa thread dump
Così vedi dove è fermo il main thread.
In main.py:
Copia codice
Python
import faulthandler
import sys

def start_ui_watchdog(timeout_s=15):
    def loop():
        last = time.time()
        while True:
            time.sleep(1)
            # se vuoi: aggiorna "heartbeat" in root.after
            if time.time() - last > timeout_s:
                logger.error("[WATCHDOG] UI frozen -> dumping threads")
                faulthandler.dump_traceback(file=sys.stderr, all_threads=True)
                last = time.time()
    threading.Thread(target=loop, daemon=True, name="UIWatchdog").start()
