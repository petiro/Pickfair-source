# controllers/dutching_controller.py
from typing import List, Dict
import time

from market_validator import MarketValidator
from dutching import (
    calculate_dutching_stakes,
    calculate_mixed_dutching
)
from ai.ai_pattern_engine import AIPatternEngine
from automation_engine import AutomationEngine
from safety_logger import SafetyLogger
from safe_mode import SafeMode


class DutchingController:
    def __init__(
        self,
        broker,
        pnl_engine,
        simulation: bool = False
    ):
        self.broker = broker
        self.pnl_engine = pnl_engine
        self.simulation = simulation

        self.ai_engine = AIPatternEngine()
        self.market_validator = MarketValidator()
        self.automation = AutomationEngine()
        self.logger = SafetyLogger()
        self.safe_mode = SafeMode()

    def submit_dutching(
        self,
        market_id: str,
        market_type: str,
        selections: List[Dict],
        total_stake: float,
        mode: str = "BACK",
        ai_enabled: bool = False,
        auto_green: bool = False
    ) -> Dict:
        """
        Entry point UNICO per tutto il dutching
        """

        # 🔒 SAFE MODE
        if self.safe_mode.is_active():
            raise RuntimeError("SAFE MODE attivo: dutching bloccato")

        # 🔎 Market validation
        if ai_enabled:
            if not self.market_validator.is_dutching_ready(market_type):
                self.logger.log_block(
                    "AI_BLOCKED_MARKET",
                    market_type=market_type
                )
                raise ValueError("Mercato non compatibile con AI dutching")

        # 🤖 AI pattern (BACK / LAY per runner)
        if ai_enabled:
            ai_sides = self.ai_engine.decide(selections)
            for sel in selections:
                sel["side"] = ai_sides.get(sel["selectionId"], "BACK")
            mode = "MIXED"

        # 🧮 Calcolo dutching (MATEMATICA INVARIATA)
        if mode == "MIXED":
            results = calculate_mixed_dutching(
                selections,
                target_profit=None,
                commission=4.5
            )
        else:
            results, _, _ = calculate_dutching_stakes(
                selections,
                total_stake,
                side=mode,
                commission=4.5
            )

        # 📈 Piazzamento ordini
        placed = []
        placed_at = time.time()

        for r in results:
            order = self.broker.place_order(
                market_id=market_id,
                selection_id=r["selectionId"],
                side=r["side"],
                price=r["price"],
                size=r["stake"],
                simulation=self.simulation
            )
            placed.append(order)

        # 🟢 Auto-green post dutching
        if auto_green and not self.simulation:
            self.automation.register_auto_green(
                market_id=market_id,
                orders=placed,
                placed_at=placed_at
            )

        return {
            "status": "OK",
            "orders": placed,
            "simulation": self.simulation
        }

# ai/ai_pattern_engine.py
class AIPatternEngine:
    """
    Decide BACK o LAY per runner basato su Weight of Money
    """

    def decide(self, selections):
        decisions = {}

        for sel in selections:
            back_vol = sum(p["size"] for p in sel.get("back_ladder", []))
            lay_vol = sum(p["size"] for p in sel.get("lay_ladder", []))

            if back_vol == 0 and lay_vol == 0:
                decisions[sel["selectionId"]] = "BACK"
                continue

            wom = back_vol / (back_vol + lay_vol)

            # Logica semplice ma efficace
            if wom > 0.55:
                decisions[sel["selectionId"]] = "BACK"
            elif wom < 0.45:
                decisions[sel["selectionId"]] = "LAY"
            else:
                decisions[sel["selectionId"]] = "BACK"

        # 🔒 Forza almeno un BACK e un LAY se possibile
        sides = set(decisions.values())
        if len(sides) == 1 and len(selections) > 1:
            first = selections[0]["selectionId"]
            decisions[first] = "LAY" if decisions[first] == "BACK" else "BACK"

        return decisions

# ui/mini_ladder.py
import customtkinter as ctk


class MiniLadder(ctk.CTkFrame):
    def __init__(self, parent, runner):
        super().__init__(parent)
        self.runner = runner
        self._build()

    def _build(self):
        ctk.CTkLabel(
            self,
            text=self.runner["runnerName"],
            anchor="w"
        ).pack(fill="x")

        frame = ctk.CTkFrame(self)
        frame.pack(fill="x")

        for side in ("BACK", "LAY"):
            col = ctk.CTkFrame(frame)
            col.pack(side="left", expand=True, fill="both")

            ladder = self.runner.get(
                "back_ladder" if side == "BACK" else "lay_ladder",
                []
            )[:3]

            best = ladder[0]["price"] if ladder else None

            for p in ladder:
                color = "#1f8b4c" if p["price"] == best else None
                ctk.CTkLabel(
                    col,
                    text=f'{p["price"]} ({p["size"]})',
                    text_color=color
                ).pack(anchor="center")

# ui/draggable_runner.py
import customtkinter as ctk


class DraggableRunner(ctk.CTkFrame):
    def __init__(self, parent, runner, on_move):
        super().__init__(parent)
        self.runner = runner
        self.on_move = on_move
        self._y = 0

        label = ctk.CTkLabel(self, text=runner["runnerName"])
        label.pack(fill="x")

        self.bind("<Button-1>", self._start)
        self.bind("<B1-Motion>", self._drag)
        self.bind("<ButtonRelease-1>", self._drop)

    def _start(self, event):
        self._y = event.y_root

    def _drag(self, event):
        dy = event.y_root - self._y
        self.place_configure(y=self.winfo_y() + dy)
        self._y = event.y_root

    def _drop(self, _):
        self.on_move(self.runner)
