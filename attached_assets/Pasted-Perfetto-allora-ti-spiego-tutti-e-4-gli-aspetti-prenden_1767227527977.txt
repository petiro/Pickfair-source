class DutchingController:
    def __init__(self, commission=4.5):
        self.commission = commission

    def run_dutching(
        self,
        ui_rows,
        mode,
        total_stake=None,
        target_profit=None
    ):
        selections = []

        for row in ui_rows:
            if not row.enabled:
                continue

            selections.append({
                "selectionId": row.selection_id,
                "runnerName": row.runner_name,
                "price": row.price,
                "side": row.side
            })

        if mode == "BACK":
            return calculate_dutching_stakes(
                selections,
                total_stake,
                "BACK",
                self.commission
            )

        if mode == "LAY":
            return calculate_dutching_stakes(
                selections,
                total_stake,
                "LAY",
                self.commission
            )

        if mode == "MIXED":
            return calculate_mixed_dutching(
                selections,
                target_profit,
                self.commission
            )

# auto_green_engine.py
import time
import threading

class AutoGreenEngine:
    def __init__(self, order_manager, tick_storage, commission=4.5):
        self.order_manager = order_manager
        self.tick_storage = tick_storage
        self.commission = commission

        self.enabled = False
        self.target_profit = None
        self.trailing = False
        self.trailing_gap = 0.0

        self._peak_pnl = {}
        self._last_action_ts = {}
        self._lock = threading.Lock()

    def enable(self, target_profit, trailing=False, trailing_gap=0.0):
        self.enabled = True
        self.target_profit = target_profit
        self.trailing = trailing
        self.trailing_gap = trailing_gap

    def disable(self):
        self.enabled = False

    def _debounce(self, market_id, seconds=2.0):
        now = time.time()
        last = self._last_action_ts.get(market_id, 0)
        if now - last < seconds:
            return False
        self._last_action_ts[market_id] = now
        return True

    def on_tick(self, market_id, positions):
        if not self.enabled:
            return

        with self._lock:
            total_pnl = sum(p["pnl"] for p in positions)

            peak = self._peak_pnl.get(market_id, total_pnl)
            self._peak_pnl[market_id] = max(peak, total_pnl)

            # 🎯 Target Profit
            if total_pnl >= self.target_profit:
                self._trigger_green(market_id, positions)
                return

            # 🔁 Trailing Stop
            if self.trailing:
                if peak - total_pnl >= self.trailing_gap:
                    self._trigger_green(market_id, positions)

    def _trigger_green(self, market_id, positions):
        if not self._debounce(market_id):
            return

        # 🔥 Liquidity check
        for p in positions:
            tick = self.tick_storage.get_last_tick(p["selectionId"])
            if not tick or tick["lay"] <= 1.01:
                return  # skip unsafe green

        self.order_manager.green_up_market(market_id)

class DutchingController:
    def __init__(self, betfair_client, order_manager, auto_green_engine):
        self.betfair = betfair_client
        self.order_manager = order_manager
        self.auto_green = auto_green_engine

    def submit_dutching(
        self,
        market_id,
        selections,
        total_stake,
        side,
        auto_green_cfg=None
    ):
        # 1️⃣ Calcolo matematico (intatto)
        if side in ("BACK", "LAY"):
            results, profit, prob = calculate_dutching_stakes(
                selections, total_stake, side
            )
        else:
            results, profit = calculate_mixed_dutching(
                selections, total_stake
            )

        # 2️⃣ Place orders Betfair
        orders = []
        for r in results:
            orders.append({
                "selectionId": r["selectionId"],
                "side": r["side"],
                "price": r["price"],
                "size": round(r["stake"], 2)
            })

        self.order_manager.place_orders(market_id, orders)

        # 3️⃣ Auto-green post dutching
        if auto_green_cfg:
            self.auto_green.enable(
                target_profit=auto_green_cfg["target"],
                trailing=auto_green_cfg["trailing"],
                trailing_gap=auto_green_cfg["gap"]
            )

        return results

class DutchingController:
    """Gestisce calcoli dutching e interfaccia con ordini Betfair"""
    def __init__(self, commission=4.5):
        self.commission = commission

    def calculate_back_dutching(self, selections, total_stake):
        sum_w = sum(1/s['price'] for s in selections)
        results = []
        for s in selections:
            stake = total_stake * (1/s['price']) / sum_w
            gross = stake * s['price']
            net_profit = (gross - total_stake) * (1-self.commission/100)
            results.append({
                'selectionId': s['selectionId'],
                'runnerName': s['runnerName'],
                'price': s['price'],
                'stake': stake,
                'profitIfWins': net_profit
            })
        return results

        # Placeholder: qui integri Betfair API
        for r in results:
            print(f"Placing {side} order: {r['runnerName']} stake={r['stake']:.2f} at {r['price']:.2f}")

class AutoGreenEngine:
    """Calcola cashout automatico post-dutching"""
    def __init__(self, commission=4.5):
        self.commission = commission

    def calculate_cashout(self, dutching_results, current_prices):
        total_net = 0
        for r in dutching_results:
            sel_id = r['selectionId']
            if sel_id in current_prices:
                back = r['stake'] * r['price']
                lay = current_prices[sel_id]['lay'] * r['stake']
                profit_win = (back - lay) * (1-self.commission/100)
                total_net += profit_win
        return total_net

import customtkinter as ctk

# ==========================
# DUTCHING CONTROLLER PRO
# ==========================
class DutchingController:
    def __init__(self, commission=4.5):
        self.commission = commission

    def calculate_back_dutching(self, selections, total_stake):
        sum_w = sum(1/s['price'] for s in selections)
        results = []
        for s in selections:
            stake = total_stake * (1/s['price']) / sum_w
            gross = stake * s['price']
            net_profit = (gross - total_stake) * (1-self.commission/100)
            results.append({
                'selectionId': s['selectionId'],
                'runnerName': s['runnerName'],
                'price': s['price'],
                'stake': stake,
                'profitIfWins': net_profit,
                'side': 'BACK'
            })
        return results

    def calculate_lay_dutching(self, selections, total_liability):
        sum_w = sum(1/(s['price']-1) for s in selections)
        results = []
        for s in selections:
            liability = total_liability * (1/(s['price']-1)) / sum_w
            stake = liability / (s['price']-1)
            net_profit = stake - liability * self.commission/100
            results.append({
                'selectionId': s['selectionId'],
                'runnerName': s['runnerName'],
                'price': s['price'],
                'stake': stake,
                'profitIfWins': net_profit,
                'side': 'LAY'
            })
        return results

    def calculate_mixed_dutching(self, selections, target_profit):
        n = len(selections)
        A = np.zeros((n,n))
        for i in range(n):
            for j in range(n):
                if i==j:
                    A[i,j] = selections[j]['price'] if j%2==0 else - (selections[j]['price']-1)
                else:
                    A[i,j] = 0
        b = np.array([target_profit]*n)
        stakes = np.linalg.solve(A, b)
        results = []
        for i, s in enumerate(selections):
            results.append({
                'selectionId': s['selectionId'],
                'runnerName': s['runnerName'],
                'price': s['price'],
                'stake': stakes[i],
                'profitIfWins': target_profit,
                'side': 'BACK' if i%2==0 else 'LAY'
            })
        return results

    def place_orders(self, results):
        for r in results:
            print(f"[ORDER] {r['side']} {r['runnerName']} stake={r['stake']:.2f} @ {r['price']:.2f}")

class AutoGreenEngine:
    def __init__(self, commission=4.5):

        total_net = 0
        for r in dutching_results:
            sel_id = r['selectionId']
            if sel_id in current_prices:
                price = current_prices[sel_id][r['side'].lower()]
                stake = r['stake']
                if r['side']=='BACK':
                    profit_win = (stake * price - stake) * (1-self.commission/100)
                else:  # LAY
                    liability = stake * (price -1)
                    profit_win = (stake - liability) * (1-self.commission/100)
                total_net += profit_win
        return total_net

    def _build_ui(self):
        toggle_frame = ctk.CTkFrame(self)
        toggle_frame.pack(fill="x", pady=2)
        ctk.CTkCheckBox(toggle_frame, text="Auto-Green", variable=self.auto_green_enabled).pack(side="left", padx=5)
        ctk.CTkCheckBox(toggle_frame, text="Book Warning", variable=self.book_warning_enabled).pack(side="left", padx=5)

        self.rows = []
        for s in self.markets:
            row = ctk.CTkFrame(self)
            row.pack(fill="x", pady=1)
            label = ctk.CTkLabel(row, text=s['runnerName'], width=60)
            label.pack(side="left")
            price_label = ctk.CTkLabel(row, text=f"{s['price']:.2f}", width=40)
            price_label.pack(side="left")
            row.labels = (label, price_label)

            for pct in [0.25,0.5,1.0]:
                btn = ctk.CTkButton(row, text=f"{int(pct*100)}%", width=40,
                                     command=lambda s=s,p=pct: self._handle_click(s,p))
                btn.pack(side="left", padx=2)

            row.bind("<Button-1>", self._on_drag_start)
            row.bind("<B1-Motion>", self._on_drag_motion)
            row.bind("<ButtonRelease-1>", self._on_drag_release)
            self.rows.append(row)

        self.book_label = ctk.CTkLabel(self, text="", text_color="red")
        self.book_label.pack(pady=2)
        self.net_label = ctk.CTkLabel(self, text="P&L Netto: 0.00€", text_color="green")
        self.net_label.pack(pady=2)

        self.fig, self.ax = plt.subplots(figsize=(6,3))
        self.canvas = FigureCanvasTkAgg(self.fig, master=self)
        self.canvas.get_tk_widget().pack(fill="both", expand=True)

    # -----------------------------
    def _handle_click(self, sel, pct):
        total_stake = 100 * pct
        results = self.dutching.calculate_back_dutching(self.markets, total_stake)
        results += self.dutching.calculate_lay_dutching(self.markets, total_stake)  # mix BACK/LAY
        self.dutching.place_orders(results)
        if self.auto_green_enabled.get():
            current_prices = {s['selectionId']: {'back': s['price']*1.05, 'lay': s['price']*0.95} for s in self.markets}
            net = self.auto_green.calculate_cashout(results, current_prices)
            self.net_label.configure(text=f"P&L Netto: {net:.2f}€")
        if self.book_warning_enabled.get():
            book = self.book_opt.check_book(self.markets)
            if book["warning"]:
                self.book_label.configure(text=f"⚠ Book %: {book['book_total']*100:.1f}%")
            else:
                self.book_label.configure(text="")

    # -----------------------------
    def _on_drag_start(self, event):
        self._drag_data["widget"] = event.widget
        self._drag_data["y"] = event.y_root
    def _on_drag_motion(self, event):
        dy = event.y_root - self._drag_data["y"]
        widget = self._drag_data["widget"]
        widget.place(y=widget.winfo_y() + dy)
        self._drag_data["y"] = event.y_root
    def _on_drag_release(self, event):
        self._drag_data["widget"] = None

    # -----------------------------
    def _start_ai_monitor(self):
        def monitor():
            while True:
                time.sleep(1)
                for sel in self.markets:
                    history = self.tick_history[sel['selectionId']]
                    action = self.ai_engine.detect_patterns(history)
                    if action:
                        print(f"AI Pattern detected {action} on {sel['runnerName']}")
        t = threading.Thread(target=monitor, daemon=True)
        t.start()

    # -----------------------------
    def update_tick(self, selectionId, price):
        self.tick_history[selectionId].append(price)
        if len(self.tick_history[selectionId])>50:
            self.tick_history[selectionId].pop(0)
        self._refresh_plot()

    def _init_plot(self):
        self.ax.clear()
        self.ax.set_title("Historic Tick Replay")
        self.ax.set_xlabel("Time")
        self.ax.set_ylabel("Price")

    def _refresh_plot(self):
        self.ax.clear()
        self.ax.set_title("Historic Tick Replay Multi-Mercato")
        self.ax.set_xlabel("Time")
        self.ax.set_ylabel("Price")
        for sel_id, history in self.tick_history.items():
            self.ax.plot(history, label=f"Sel {sel_id}")
        self.ax.legend()
        self.canvas.draw()
