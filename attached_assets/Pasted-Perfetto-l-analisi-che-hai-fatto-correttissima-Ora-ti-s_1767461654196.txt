def _on_tree_click(self, event):
    item_id = self.tree.identify_row(event.y)
    if not item_id:
        return

    item_type = self.tree.set(item_id, "type")
    if item_type != "market":
        return

    market_id = self.tree.set(item_id, "market_id")
    if not market_id:
        return

    self.load_market(market_id)
👉 1 click = 1 market load 👉 Zero dipendenza dallo stato interno del tree
