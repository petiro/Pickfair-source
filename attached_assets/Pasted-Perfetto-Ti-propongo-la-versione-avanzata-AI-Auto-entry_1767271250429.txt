        # Step 4: Se non in simulazione, puoi inviare submit automatico
        if not self.simulation_mode:
            self.place_orders(stakes_info)

    def calculate_mixed_stakes(self, selections, target_profit=50.0):
        """
        Calcola stake misti BACK+LAY per garantire profitto uniforme
        """
        n = len(selections)
        prices = np.array([s['price'] for s in selections])

        # Decide BACK o LAY automaticamente: BACK se < prob. implied media, altrimenti LAY
        implied_probs = 1 / prices
        avg_prob = np.mean(implied_probs)
        sides = ['BACK' if p < avg_prob else 'LAY' for p in implied_probs]

        # Costruisci sistema lineare: A * stakes = b
        A = np.zeros((n, n))
        b = np.ones(n) * target_profit
        for i in range(n):
            for j in range(n):
                if sides[j] == 'BACK':
                    A[i, j] = prices[j] if i == j else 0
                else:  # LAY
                    A[i, j] = -(prices[j]-1) if i == j else 0

        # Risolvi sistema lineare per stake
        try:
            stakes = np.linalg.solve(A, b)
        except np.linalg.LinAlgError:
            # fallback: distribuzione uniforme
            stakes = np.ones(n) * target_profit / np.mean(prices)

        # Costruisci info completo
        stakes_info = []
        for s, side, stake in zip(selections, sides, stakes):
            stakes_info.append({
                'selectionId': s['selectionId'],
                'runnerName': s['runnerName'],
                'side': side,
                'stake': max(stake, 0.0),  # evita negativi
                'price': s['price']
            })

        # Calcolo profitto netto stimato
        profits = []
        for s in stakes_info:
            if s['side'] == 'BACK':
                profits.append(s['stake'] * (s['price'] - 1))
            else:
                profits.append(s['stake'])
        uniform_profit = min(profits)
        return stakes_info, uniform_profit

class DutchingDialogProDragDrop(DutchingDialogProAdvanced):
    def __init__(self):
        super().__init__()
        self.drag_data = {"widget": None, "y": 0}
        self._create_ladder_widgets()

    def _create_ladder_widgets(self):
        # Creiamo runner widgets con possibilità di drag & drop
        for i, runner in enumerate(self.runner_widgets):
            widget = runner['frame']
            widget.bind("<ButtonPress-1>", self.on_start_drag)
            widget.bind("<ButtonRelease-1>", self.on_drop)
            widget.bind("<B1-Motion>", self.on_drag_motion)

    # ---------------- Drag & Drop Handlers ----------------
    def on_start_drag(self, event):
        self.drag_data["widget"] = event.widget
        self.drag_data["y"] = event.y_root

    def on_drag_motion(self, event):
        dy = event.y_root - self.drag_data["y"]
        event.widget.place_configure(y=event.widget.winfo_y() + dy)
        self.drag_data["y"] = event.y_root

    def on_drop(self, event):
        dropped_widget = self.drag_data["widget"]
        self.drag_data["widget"] = None

        # Determina nuovo livello in base alla posizione Y
        new_level = int(event.y_root / 50)  # ogni 50px è un livello
        new_level = min(max(new_level, 0), self.ladder_levels-1)
        dropped_widget.grid(row=new_level, column=0)

        # Ricalcola stake automaticamente
        self.ai_mixed_auto_entry()

import time
from collections import deque

class TickReplayEngine:
    def __init__(self, ladder_dialog, tick_history, speed=1.0):
        """
        ladder_dialog: istanza di DutchingDialogProDragDrop
        tick_history: lista di tick storici [{'selectionId':1,'back':2.1,'lay':2.3}, ...]
        speed: moltiplicatore tempo, 1.0 = reale, 2.0 = doppio velocità
        """
        self.ladder_dialog = ladder_dialog
        self.tick_history = deque(tick_history)
        self.speed = speed
        self.running = False

    def start_replay(self):
        self.running = True
        self._run()

    def stop_replay(self):
        self.running = False

    def _run(self):
        while self.tick_history and self.running:
            tick = self.tick_history.popleft()
            self._apply_tick(tick)
            # Delay in base a speed
            time.sleep(1.0 / self.speed)

    def _apply_tick(self, tick):
        # Aggiorna quote nella ladder
        self.ladder_dialog.update_runner_price(
            tick['selectionId'], back=tick.get('back'), lay=tick.get('lay')
        )
        # Ricalcola automaticamente il Dutching
        self.ladder_dialog.ai_mixed_auto_entry()
        # Eventuale Auto-Green post Dutching
        self.ladder_dialog.auto_green_check()

class DraggableRunner(ctk.CTkFrame):
    def __init__(self, parent, runner_data, ladder_dialog):
        super().__init__(parent, fg_color="gray20", corner_radius=5)
        self.runner_data = runner_data
        self.ladder_dialog = ladder_dialog
        self.pack(pady=2, fill="x")

        # Label runner
        self.label = ctk.CTkLabel(self, text=f"{runner_data['runnerName']} | {runner_data['price']}")
        self.label.pack(side="left", padx=5)

        # Bind drag events
        self.label.bind("<Button-1>", self.start_drag)
        self.label.bind("<B1-Motion>", self.drag_motion)
        self.label.bind("<ButtonRelease-1>", self.drop_runner)

        self.start_y = 0

    def start_drag(self, event):
        self.start_y = event.y_root

    def drag_motion(self, event):
        dy = event.y_root - self.start_y
        self.place_configure(y=self.winfo_y() + dy)
        self.start_y = event.y_root

    def drop_runner(self, event):
        # Ricalcola posizione nell'elenco
        self.ladder_dialog.reorder_runners(self)
        # Aggiorna stakes e Auto-Green
        self.ladder_dialog.recalculate_dutching()
        if self.ladder_dialog.auto_green_enabled:
            self.ladder_dialog.auto_green_engine.apply_auto_green()

class DutchingController:
    def __init__(self, ladder_dialog):
        self.ladder_dialog = ladder_dialog
        self.runners = ladder_dialog.runners

    def reorder_runners(self, dragged_runner):
        # Ordina lista in base alla posizione visiva
        self.runners.sort(key=lambda r: r.winfo_y())
        self.ladder_dialog.runners = self.runners
        self.recalculate_dutching()

    def recalculate_dutching(self):
        selections = [r.runner_data for r in self.runners]
        total_stake = self.ladder_dialog.get_total_stake()
        results, profit, _ = calculate_dutching_stakes(selections, total_stake, side="BACK")
        for r, res in zip(self.runners, results):
            r.label.configure(text=f"{res['runnerName']} | {res['stake']:.2f} €")

class AutoGreenEngine:
    def __init__(self, ladder_dialog):
        self.ladder_dialog = ladder_dialog

    def apply_auto_green(self):
        # Calcola cashout ottimale
        for runner in self.ladder_dialog.runners:
            back = runner.runner_data['price']
            lay = self.ladder_dialog.get_current_lay(runner.runner_data['selectionId'])
            if lay:
                profit_win, profit_lose = dynamic_cashout_single(
                    back_stake=runner.runner_data['stake'],
                    back_price=back,
                    lay_price=lay,
                    commission=COMMISSION
                )
                # Auto-cashout se profitto garantito positivo
                if min(profit_win, profit_lose) > 0:
                    runner.label.configure(fg_color="green")
                else:
                    runner.label.configure(fg_color="gray20")
