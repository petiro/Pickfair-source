📈 Equity curve (cumulativa)
Copia codice

def get_equity_curve():
    with get_db() as conn:
        rows = conn.execute("""
        SELECT
            timestamp,
            profit
        FROM bet_history
        WHERE status IN ('CASHED_OUT', 'SETTLED_WIN', 'SETTLED_LOSS')
        ORDER BY timestamp
        """).fetchall()

    equity = 0
    curve = []
    for ts, profit in rows:
        equity += profit
        curve.append({
            "timestamp": ts,
            "equity": round(equity, 2)
        })
    return curve

def get_dashboard_kpis():
    with get_db() as conn:
        row = conn.execute("""
        SELECT
            COUNT(*) as bets,
            ROUND(SUM(profit), 2) as pnl,
            ROUND(AVG(profit), 2) as avg_profit,
            ROUND(100.0 * SUM(CASE WHEN profit > 0 THEN 1 ELSE 0 END) / COUNT(*), 2) as winrate
        FROM bet_history
        WHERE status IN ('CASHED_OUT', 'SETTLED_WIN', 'SETTLED_LOSS')
        """).fetchone()

    return {
        "bets": row[0],
        "pnl": row[1],
        "avg_profit": row[2],
        "winrate": row[3]
    }

def get_telegram_history(limit=200):
    with get_db() as conn:
        rows = conn.execute("""
        SELECT
            timestamp,
            bet_id,
            status,
            message
        FROM telegram_audit
        ORDER BY timestamp DESC
        LIMIT ?
        """, (limit,)).fetchall()

    return rows

import matplotlib.pyplot as plt
from matplotlib.backends.backend_tkagg import FigureCanvasTkAgg

def render_equity_chart(parent):
    data = get_equity_curve()

    x = [d['timestamp'] for d in data]
    y = [d['equity'] for d in data]

    fig, ax = plt.subplots(figsize=(6,3))
    ax.plot(y)
    ax.set_title("Equity Curve")
    ax.set_ylabel("€")
    ax.grid(True)

    canvas = FigureCanvasTkAgg(fig, master=parent)
    canvas.draw()
    canvas.get_tk_widget().pack(fill="both", expand=True)

def render_telegram_table(parent):
    rows = get_telegram_history()

    tree = ttk.Treeview(
        parent,
        columns=("time", "bet", "status", "msg"),
        show="headings"
    )

    for col in tree["columns"]:
        tree.heading(col, text=col.upper())

    for r in rows:
        tree.insert("", "end", values=r)

    tree.pack(fill="both", expand=True)
