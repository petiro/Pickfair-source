# dutching_controller.py
import numpy as np
from threading import Lock

class DutchingController:
    def __init__(self, commission=4.5):
        self.commission = commission
        self.simulation_mode = True  # True = simulazione, False = live
        self.orders = []
        self.lock = Lock()

    # Entry point per calcolo dutching
    def calculate_dutching(self, selections, total_stake, side="BACK"):
        if side.upper() == "BACK":
            return self._calculate_back_dutching(selections, total_stake)
        elif side.upper() == "LAY":
            return self._calculate_lay_dutching(selections, total_stake)
        else:
            raise ValueError("Side must be BACK or LAY")

    # BACK dutching
    def _calculate_back_dutching(self, selections, total_stake):
        weights = [1/s['price'] for s in selections]
        sum_w = sum(weights)
        results = []
        for s, w in zip(selections, weights):
            stake = total_stake * w / sum_w
            gross_profit = stake * s['price'] - total_stake
            net_profit = gross_profit * (1 - self.commission/100)
            results.append({
                'selectionId': s['selectionId'],
                'runnerName': s['runnerName'],
                'price': s['price'],
                'stake': stake,
                'grossProfit': gross_profit,
                'profitIfWins': net_profit,
                'side': 'BACK'
            })
        uniform_profit = results[0]['profitIfWins'] if results else 0
        implied_probability = sum(1/s['price'] for s in selections)
        return results, uniform_profit, implied_probability

    # LAY dutching
    def _calculate_lay_dutching(self, selections, total_liability):
        weights = [1/(s['price']-1) for s in selections]
        sum_w = sum(weights)
        results = []
        for s, w in zip(selections, weights):
            liability = total_liability * w / sum_w
            stake = liability / (s['price'] - 1)
            gross_profit = stake  # semplificato
            net_profit = gross_profit * (1 - self.commission/100)
            results.append({
                'selectionId': s['selectionId'],
                'runnerName': s['runnerName'],
                'price': s['price'],
                'stake': stake,
                'grossProfit': gross_profit,
                'profitIfWins': net_profit,
                'side': 'LAY'
            })
        uniform_profit = results[0]['profitIfWins'] if results else 0
        implied_probability = sum(1/(s['price']-1) for s in selections)
        return results, uniform_profit, implied_probability

    # Submit ordini: live vs simulazione
    def place_orders(self, orders):
        with self.lock:
            if self.simulation_mode:
                # Simulazione: salviamo solo in memoria
                for o in orders:
                    o['status'] = 'SIMULATED'
                self.orders.extend(orders)
                print(f"[SIMULAZIONE] {len(orders)} ordini calcolati e salvati")
            else:
                # Live: invio a Betfair API
                for o in orders:
                    o['status'] = 'SUBMITTED'
                self.orders.extend(orders)
                print(f"[LIVE] {len(orders)} ordini inviati a Betfair")
            return orders

    # Auto-green post dutching
    def auto_green(self, orders, market_prices):
        green_orders = []
        for o in orders:
            side = o['side']
            stake = o['stake']
            price = o['price']
            if side == 'BACK':
                lay_price = market_prices.get(o['runnerName'], price)
                lay_stake = (stake * price) / lay_price
                profit_win = stake * (price - 1) - lay_stake * (lay_price - 1)
                profit_lose = lay_stake - stake
            else:  # LAY
                back_price = market_prices.get(o['runnerName'], price)
                back_stake = (stake * (price -1)) / (back_price -1)
                profit_win = stake
                profit_lose = back_stake * (back_price -1) - stake
            net_profit = min(profit_win, profit_lose) * (1 - self.commission/100)
            green_orders.append({
                'selectionId': o['selectionId'],
                'runnerName': o['runnerName'],
                'netProfit': net_profit,
                'side': side
            })
        return green_orders

import numpy as np

class DutchingDialogPro(ctk.CTk):
    def __init__(self):
        super().__init__()
        # ... codice precedente ...
        self.target_profit = 50.0  # Target netto automatico per AI
        # Bottone AI auto-entry
        self.ai_btn = ctk.CTkButton(self, text="AI Auto-entry", command=self.ai_auto_entry)
        self.ai_btn.pack(pady=5)

    # --- AI Pattern Auto-entry ---
    def ai_auto_entry(self):
        # Step 1: recupera selezioni
        selections = []
        for w in self.runner_widgets:
            price = w['runner']['price']
            selections.append({
                'selectionId': w['runner']['selectionId'],
                'runnerName': w['runner']['runnerName'],
                'price': price
            })

        # Step 2: calcolo stake ottimale BACK per target profit
        stakes, uniform_profit = self.calculate_ai_stakes(selections, target_profit=self.target_profit)

        # Step 3: aggiorna UI
        for w, stake in zip(self.runner_widgets, stakes):
            w['stake_entry'].delete(0, "end")
            w['stake_entry'].insert(0, f"{stake:.2f}")
        self.pl_preview.configure(text=f"Net Profit Preview: {uniform_profit:.2f} €")

    def calculate_ai_stakes(self, selections, target_profit=50.0):
        """
        Calcola stake ottimale per target profit netto su tutte le selezioni (BACK)
        """
        prices = np.array([s['price'] for s in selections])
        weights = 1 / prices
        sum_w = np.sum(weights)
        total_stake = target_profit / ((1/sum_w) - 1e-6)  # evita divisione zero
        stakes = total_stake * weights / sum_w

        # Calcolo profitto netto simulato
        gross_profits = stakes * prices
        uniform_profit = np.min(gross_profits - total_stake)  # netto min
        return stakes, uniform_profit
