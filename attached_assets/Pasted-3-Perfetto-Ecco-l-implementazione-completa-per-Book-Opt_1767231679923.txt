class DraggableRunner(ctk.CTkFrame):
    def __init__(self, parent, runner_data, ladder_dialog):
        super().__init__(parent, corner_radius=5, border_width=1, border_color="gray")
        self.runner_data = runner_data
        self.ladder_dialog = ladder_dialog
        self.label = ctk.CTkLabel(self, text=self.get_label_text())
        self.label.pack(padx=5, pady=5, fill='x')
        self.pack(pady=2, fill='x')

        # Drag & Drop bindings
        self.label.bind("<Button-1>", self.on_start)
        self.label.bind("<B1-Motion>", self.on_drag)
        self.label.bind("<ButtonRelease-1>", self.on_drop)
        self._drag_start_y = 0

    def get_label_text(self):
        stake = self.runner_data.get('stake',0)
        price = self.runner_data.get('price',0)
        return f"{self.runner_data['runnerName']} | {price:.2f} € | Stake {stake:.2f} €"

    def on_start(self, event):
        self._drag_start_y = event.y_root

    def on_drag(self, event):
        dy = event.y_root - self._drag_start_y
        self.place_configure(y=self.winfo_y() + dy)
        self._drag_start_y = event.y_root

    def on_drop(self, event):
        # Ricalcola ordinamento
        self.ladder_dialog.reorder_runners()

class DutchingReplayDialog(DutchingDialog):
    def __init__(self, parent):
        super().__init__(parent)

        # Toggle Auto-Green
        self.auto_green_var = ctk.BooleanVar(value=True)
        ctk.CTkCheckBox(self, text="Auto-Green ON/OFF", variable=self.auto_green_var,
                        command=self.toggle_auto_green).pack(pady=5)

        # Replay Frame
        self.replay_frame = ctk.CTkFrame(self)
        self.replay_frame.pack(fill='both', expand=True, pady=5)

        # Inline Chart
        self.figure = Figure(figsize=(5,2), dpi=100)
        self.ax = self.figure.add_subplot(111)
        self.ax.set_title("Profitto Netto Replay Tick")
        self.ax.set_xlabel("Tick")
        self.ax.set_ylabel("Profitto €")
        self.line, = self.ax.plot([], [], color='blue')
        self.canvas = FigureCanvasTkAgg(self.figure, master=self.replay_frame)
        self.canvas.get_tk_widget().pack(fill='both', expand=True)

        self.replay_data = []  # [(tick, profit)]
        self.replay_thread = None

    def toggle_auto_green(self):
        self.auto_green_enabled = self.auto_green_var.get()
        if self.auto_green_enabled:
            self.auto_green_engine.apply_auto_green()
        else:
            # reset colori
            for r in self.runners:
                r.label.configure(fg_color="")

    def add_tick_to_replay(self, tick_number, profit):
        self.replay_data.append((tick_number, profit))
        ticks, profits = zip(*self.replay_data)
        self.line.set_data(ticks, profits)
        self.ax.relim()
        self.ax.autoscale_view()
        self.canvas.draw()

    def start_replay_simulation(self, interval=0.5, max_ticks=50):
        def replay_loop():
            for tick in range(max_ticks):
                # Simula profitto netto casuale dai runner
                profit = sum(r.runner_data.get('profitIfWins',0) for r in self.runners)
                profit += random.uniform(-2,2)  # rumore tick
                self.add_tick_to_replay(tick, profit)
                time.sleep(interval)
        self.replay_thread = threading.Thread(target=replay_loop, daemon=True)
        self.replay_thread.start()

class BookOptimizer:
    def __init__(self, warning_threshold=105.0):
        self.warning_threshold = warning_threshold

    def calculate_book(self, selections):
        if not selections:
            return None, None

        book = sum(1 / s['price'] for s in selections if s['price'] > 1.01) * 100
        status = (
            "GOOD" if book < 100
            else "WARNING" if book <= self.warning_threshold
            else "BAD"
        )
        return round(book, 2), status

        self.wom_threshold = wom_threshold

    def analyze(self, ladder_levels):
        back_vol = sum(l['back_vol'] for l in ladder_levels)
        lay_vol = sum(l['lay_vol'] for l in ladder_levels)

        total = back_vol + lay_vol
        if total == 0:
            return None

        wom = (back_vol / total) * 100

        if wom > self.wom_threshold:
            return "BACK"
        elif wom < (100 - self.wom_threshold):
            return "LAY"
        return None

class TickReplayEngine:
    def __init__(self):
        self.ticks = []

    def add_tick(self, price, volume, timestamp):
        self.ticks.append((price, volume, timestamp))

    def replay(self, callback, speed=0.2):
        for tick in self.ticks:
            callback(tick)
            time.sleep(speed)

def calculate_book_percentage(runners):
    probs = []
    for r in runners:
        price = r.get("bestBack")
        if price and price > 1.01:
            probs.append(1 / price)
    return round(sum(probs) * 100, 2)

class DutchingController:

    def __init__(self, betfair_client, simulation=False):
        self.client = betfair_client
        self.simulation = simulation

    def submit_dutching(self, selections, stake, side, auto_green=False):
        from dutching import calculate_dutching_stakes

        results, profit, _ = calculate_dutching_stakes(
            selections=selections,
            total_stake=stake,
            side=side,
            commission=4.5
        )

        if self.simulation:
            return self._simulate(results)

        order_ids = []
        for r in results:
            resp = self.client.place_order(
                selection_id=r['selectionId'],
                side=r['side'],
                price=r['price'],
                size=r['stake']
            )
            order_ids.append(resp['betId'])

        if auto_green:
            AutoGreenEngine.attach(order_ids)

        return order_ids

    def _simulate(self, results):
        for r in results:
            r['simulated'] = True
        return results

class TrailingStopEngine:

    def __init__(self, trail_pct=30):
        self.peak = {}
        self.trail_pct = trail_pct

    def update(self, bet_id, pnl):
        peak = self.peak.get(bet_id, pnl)
        self.peak[bet_id] = max(peak, pnl)

        stop_level = peak * (1 - self.trail_pct / 100)

        if pnl < stop_level:
            return True  # trigger green
        return False
