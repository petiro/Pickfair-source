📄 storage.py
Copia codice

import sqlite3
from pathlib import Path

DB_PATH = Path("data/bets.db")
DB_PATH.parent.mkdir(exist_ok=True)

def get_db():
    return sqlite3.connect(DB_PATH, check_same_thread=False)

def init_db():
    with get_db() as conn:
        conn.execute("""
        CREATE TABLE IF NOT EXISTS bet_history (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            bet_id TEXT,
            market_id TEXT,
            selection_id TEXT,
            side TEXT,
            stake REAL,
            price REAL,
            matched_size REAL,
            status TEXT,
            profit REAL,
            error_code TEXT,
            error_message TEXT,
            timestamp DATETIME DEFAULT CURRENT_TIMESTAMP
        )
        """)
📌 Questo file NON deve mai essere cancellato

from storage import get_db

def log_bet_event(
    bet_id,
    market_id,
    selection_id,
    side,
    stake,
    price,
    status,
    matched_size=0.0,
    profit=None,
    error_code=None,
    error_message=None
):
    with get_db() as conn:
        conn.execute("""
        INSERT INTO bet_history (
            bet_id, market_id, selection_id, side,
            stake, price, matched_size,
            status, profit,
            error_code, error_message
        ) VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)
        """, (
            bet_id, market_id, selection_id, side,
            stake, price, matched_size,
            status, profit,
            error_code, error_message
        ))
